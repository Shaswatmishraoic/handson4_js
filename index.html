        2."let count = 0;
        (function () {
          if (count === 0) {
            let count = 1;
            console.log(count); // What is logged?
          }
          console.log(count); // What is logged?
        })();"

Output: 0 1

The code snippet provided defines a variable count with an initial value of 0. It then immediately invokes an anonymous self-invoking function. Within the self-invoking function, there is an if statement that checks if the outer count variable is equal to 0. If it is, a new block-scoped variable count is declared and assigned the value 1 within the scope of the if statement. After that, the value of the inner count variable is logged to the console. The first console.log(count) statement within the if statement will log 1 because it refers to the inner count variable within the block scope. The second console.log(count) statement outside the if statement will log 0 because it refers to the outer count variable in the global scope, which was not modified within the self-invoking function.

5.Take a variable in outer function and create an inner function to increase the counter every time it is called 
        function counter() {
            var count = 0; // Counter variable declared in the outer function
          
            function increaseCounter() {
              count++; // Increase the counter variable by 1
              console.log("Counter increased! Current count: " + count);
            }
          
            function resetCounter() {
              count = 0; // Reset the counter variable to 0
              console.log("Counter reset!");
            }
          
            return {
              increment: increaseCounter,
              reset: resetCounter
            };
          }
          
          var myCounter = counter(); // Outer function called and object with inner functions returned
          
          myCounter.increment(); // Output: Counter increased! Current count: 1
          myCounter.increment(); // Output: Counter increased! Current count: 2
          myCounter.reset(); // Output: Counter reset!
          myCounter.increment(); // Output: Counter increased! Current count: 1
